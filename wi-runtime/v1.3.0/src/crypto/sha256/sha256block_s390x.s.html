     9		LA	tmp-16(SP), R1
    10		XOR	R0, R0         // query function code is 0
    11		WORD    $0xB93E0006    // KIMD (R6 is ignored)
    12		MOVBZ	tmp-16(SP), R4 // get the first byte
    13		AND	$0x20, R4      // bit 2 (big endian) for SHA256
    14		CMPBEQ	R4, $0, nosha256
    15		MOVB	$1, ret+0(FP)

    18		MOVB	$0, ret+0(FP)
    19		RET

    32		RET
