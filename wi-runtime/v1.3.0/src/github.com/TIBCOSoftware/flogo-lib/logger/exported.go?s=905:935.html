  }
  
  func GetDefaultLogger() Logger {
  	defLogger := GetLogger(defaultLoggerName)
  	if defLogger == nil {
  		errorMsg := fmt.Sprintf("error getting default logger '%s'", defaultLoggerName)
  		panic(errorMsg)
  	}
  	return defLogger
  }
