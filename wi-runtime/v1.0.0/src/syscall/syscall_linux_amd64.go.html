    68	func Gettimeofday(tv *Timeval) (err error) {
    69		errno := gettimeofday(tv)
    70		if errno != 0 {
    71			return errno
    72		}
    73		return nil
    74	}

    76	func Time(t *Time_t) (tt Time_t, err error) {
    77		var tv Timeval
    78		errno := gettimeofday(&tv)
    79		if errno != 0 {
    80			return 0, errno
    81		}
    82		if t != nil {
    83			*t = Time_t(tv.Sec)
    84		}
    85		return Time_t(tv.Sec), nil
    86	}

    98	func Pipe(p []int) (err error) {
    99		if len(p) != 2 {
   100			return EINVAL
   101		}
   102		var pp [2]_C_int
   103		err = pipe(&pp)
   104		p[0] = int(pp[0])
   105		p[1] = int(pp[1])
   106		return
   107	}
