     8	// This function will check if the value is an environment value (for example {MY_VALUE})
     9	// if it is an environment value it will get resolved, otherwise the original value is returned
    10	func ResolveEnv(value string) string {
    11		if len(value) == 0 {
    12			return value
    13		}
    14		if strings.HasPrefix(value, "{") && strings.HasSuffix(value, "}") {
    15			// This is an environment value, resolving...
    16			value = value[1 : len(value)-1]
    17			if len(value) == 0 {
    18				return value
    19			}
    20			return os.Getenv(value)
    21		}
    22		return value
    23	}
