    15	type FlowBehavior interface {
    16	
    17		// Start the flow instance.  Returning true indicates that the
    18		// flow can start and eval will be scheduled on the Root Task.
    19		// Return false indicates that the flow could not be started
    20		// at this time.
    21		Start(context FlowContext) (start bool, evalCode int)
    22	
    23		// Resume the flow instance.  Returning true indicates that the
    24		// flow can resume.  Return false indicates that the flow

    26		Resume(context FlowContext) bool //<---

    30		// TasksDone is called when the RootTask is Done.

    37	// TaskBehavior is the execution behavior of a Task.
    38	type TaskBehavior interface {
    39	
    40		// Enter determines if a Task is ready to be evaluated, returning true
    41		// indicates that the task is ready to be evaluated.
    42		Enter(context TaskContext, enterCode int) (eval bool, evalCode int)
    43	
    44		// Eval is called when a Task is being evaluated.  Returning true indicates
    45		// that the task is done.  If err is set, it indicates that the
    46		// behavior intends for the flow ErrorHandler to handle the error
    47		Eval(context TaskContext, evalCode int) (done bool, doneCode int, err error)
    48	
    49		// PostEval is called when a task that didn't complete during the Eval
    50		// needs to be notified.  Returning true indicates that the task is done.
    51		// If err is set, it indicates that the  behavior intends for the
    52		// flow ErrorHandler to handle the error
    53		PostEval(context TaskContext, evalCode int, data interface{}) (done bool, doneCode int, err error)
    54	
    55		// Done is called when Eval, PostEval or ChildDone return true, indicating
    56		// that the task is done.  This step is used to finalize the task and
    57		// determine the next set of tasks to be entered.  Returning true indicates
    58		// that the parent task should be notified.  Also returns the set of Tasks
    59		// that should be entered next.
    60		Done(context TaskContext, doneCode int) (notifyParent bool, childDoneCode int, taskEntries []*TaskEntry, err error)
    61	
    62		// Error is called when there is an issue executing Eval, it returns a boolean indicating
    63		// if it handled the error, otherwise the error is handled by the global error handler
    64		Error(context TaskContext) (handled bool, taskEntry *TaskEntry)
    65	
    66		// ChildDone is called when child task is Done and has indicated that its
    67		// parent should be notified.  Returning true indicates that the task
    68		// is done.
    69		ChildDone(context TaskContext, childTask *definition.Task, childDoneCode int) (done bool, doneCode int)
    70	}
