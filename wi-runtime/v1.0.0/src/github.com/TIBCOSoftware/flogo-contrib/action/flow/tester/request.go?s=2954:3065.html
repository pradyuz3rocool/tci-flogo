    35	// StartFlow handles a StartRequest for a FlowInstance.  This will
    36	// generate an ID for the new FlowInstance and queue a StartRequest.
    37	func (rp *RequestProcessor) StartFlow(startRequest *StartRequest) (code int, retData interface{}, err error) {
    38	
    39		execOptions := &instance.ExecOptions{Interceptor: startRequest.Interceptor, Patch: startRequest.Patch}
    40	
    41		attrs := startRequest.Attrs
    42	
    43		dataLen := len(startRequest.Data)
    44	
    45		// attrs, not supplied so attempt to create attrs from Data
    46		if attrs == nil && dataLen > 0 {
    47			attrs = make([]*data.Attribute, 0, dataLen)
    48	
    49			for k, v := range startRequest.Data {
    50	
    51				//todo handle error
    52				t, _ := data.GetType(v)
    53				attrs = append(attrs, data.NewAttribute(k, t, v))
    54			}
    55		}
    56	
    57		factory := action.GetFactory(FLOW_REF)
    58		act := factory.New(&action.Config{Id:"flow"})
    59	
    60		ctx := trigger.NewContext(context.Background(), attrs)
    61	
    62		ro := &instance.RunOptions{Op: instance.OpStart, ReturnID: true, ExecOptions: execOptions}
    63		return rp.runner.Run(ctx, act, startRequest.FlowURI, ro)
    64	}

    66	// RestartFlow handles a RestartRequest for a FlowInstance.  This will
    67	// generate an ID for the new FlowInstance and queue a RestartRequest.
    68	func (rp *RequestProcessor) RestartFlow(restartRequest *RestartRequest) (code int, retData interface{}, err error) {
    69	
    70		execOptions := &instance.ExecOptions{Interceptor: restartRequest.Interceptor, Patch: restartRequest.Patch}
    71	
    72		ctx := context.Background()
    73	
    74		if restartRequest.Data != nil {
    75	
    76			logger.Debugf("Updating flow attrs: %v", restartRequest.Data)
    77			attrs := make([]*data.Attribute, len(restartRequest.Data))
    78	
    79			for k, v := range restartRequest.Data {
    80				attrs = append(attrs, data.NewAttribute(k, data.ANY, v))
    81			}
    82	
    83			ctx = trigger.NewContext(context.Background(), attrs)
    84		}
    85	
    86		factory := action.GetFactory(FLOW_REF)
    87		act := factory.New(&action.Config{Id:"flow"})
    88	
    89		ro := &instance.RunOptions{Op: instance.OpRestart, ReturnID: true, InitialState: restartRequest.InitialState, ExecOptions: execOptions}
    90		return rp.runner.Run(ctx, act, restartRequest.InitialState.FlowURI, ro)
    91	}
