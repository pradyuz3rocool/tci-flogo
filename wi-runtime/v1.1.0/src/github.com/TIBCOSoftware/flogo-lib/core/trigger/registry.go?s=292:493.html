    51	func (r *registry) AddFactory(ref string, f Factory) error {
    52		triggersMu.Lock()
    53		defer triggersMu.Unlock()
    54	
    55		logger.Debugf("Registering trigger factory: '%s'", ref)
    56	
    57		if len(ref) == 0 {
    58			return fmt.Errorf("registry.RegisterFactory: ref is empty")
    59		}
    60	
    61		if f == nil {
    62			return fmt.Errorf("registry.RegisterFactory: factory is nil")
    63		}
    64	
    65		// copy on write to avoid synchronization on access
    66		newFs := make(map[string]Factory, len(r.factories))
    67	
    68		for k, v := range r.factories {
    69			newFs[k] = v
    70		}
    71	
    72		if newFs[ref] != nil {
    73			return fmt.Errorf("registry.RegisterFactory: already registered factory for ref '%s'", ref)
    74		}
    75	
    76		newFs[ref] = f
    77	
    78		r.factories = newFs
    79	
    80		return nil
    81	}

   133	// Register registers the specified trigger
   134	func Register(trigger TriggerDeprecated) {
   135		triggersMu.Lock()
   136		defer triggersMu.Unlock()
   137	
   138		if trigger == nil {
   139			panic("trigger.Register: trigger is nil")
   140		}
   141		id := trigger.Metadata().ID
   142	
   143		if _, dup := triggers[id]; dup {
   144			panic("trigger.Register: Register called twice for trigger " + id)
   145		}
   146		// copy on write to avoid synchronization on access
   147		newTriggers := make(map[string]TriggerDeprecated, len(triggers))
   148	
   149		for k, v := range triggers {
   150			newTriggers[k] = v
   151		}
   152	
   153		newTriggers[id] = trigger
   154		triggers = newTriggers
   155	}
