    67	// Container - an internal structure that holds a reference to the core interface map of the parsed
    68	// json. Use this container to move context.
    69	type Container struct {
    70		object interface{}
    71	}
    72	
    73	// Data - Return the contained data as an interface{}.
    74	func (g *Container) Data() interface{} {
    75		return g.object
    76	}

    80	// Path - Search for a value using dot notation.
    81	func (g *Container) Path(path string) *Container {
    82		return g.Search(strings.Split(path, ".")...)
    83	}
    84	
    85	// Search - Attempt to find and return an object within the JSON structure by specifying the
    86	// hierarchy of field names to locate the target. If the search encounters an array and has not
    87	// reached the end target then it will iterate each object of the array for the target and return
    88	// all of the results in a JSON array.
    89	func (g *Container) Search(hierarchy ...string) *Container {
    90		var object interface{}
    91	
    92		object = g.object
    93		for target := 0; target < len(hierarchy); target++ {
    94			if mmap, ok := object.(map[string]interface{}); ok {
    95				object = mmap[hierarchy[target]]
    96			} else if marray, ok := object.([]interface{}); ok {
    97				tmpArray := []interface{}{}
    98				for _, val := range marray {
    99					tmpGabs := &Container{val}
   100					res := tmpGabs.Search(hierarchy[target:]...).Data()
   101					if res != nil {
   102						tmpArray = append(tmpArray, res)
   103					}
   104				}
   105				if len(tmpArray) == 0 {
   106					return &Container{nil}
   107				}
   108				return &Container{tmpArray}
   109			} else {
   110				return &Container{nil}
   111			}
   112		}
   113		return &Container{object}
   114	}

   131	// Index - Attempt to find and return an object within a JSON array by index.
   132	func (g *Container) Index(index int) *Container {
   133		if array, ok := g.Data().([]interface{}); ok {
   134			if index >= len(array) {
   135				return &Container{nil}
   136			}
   137			return &Container{array[index]}
   138		}
   139		return &Container{nil}
   140	}

   142	// Children - Return a slice of all the children of the array. This also works for objects, however,
   143	// the children returned for an object will NOT be in order and you lose the names of the returned
   144	// objects this way.
   145	func (g *Container) Children() ([]*Container, error) {
   146		if array, ok := g.Data().([]interface{}); ok {
   147			children := make([]*Container, len(array))
   148			for i := 0; i < len(array); i++ {
   149				children[i] = &Container{array[i]}
   150			}
   151			return children, nil
   152		}
   153		if mmap, ok := g.Data().(map[string]interface{}); ok {
   154			children := []*Container{}
   155			for _, obj := range mmap {
   156				children = append(children, &Container{obj})
   157			}
   158			return children, nil
   159		}
   160		return nil, ErrNotObjOrArray
   161	}
   162	
   163	// ChildrenMap - Return a map of all the children of an object.
   164	func (g *Container) ChildrenMap() (map[string]*Container, error) {
   165		if mmap, ok := g.Data().(map[string]interface{}); ok {
   166			children := map[string]*Container{}
   167			for name, obj := range mmap {
   168				children[name] = &Container{obj}
   169			}
   170			return children, nil
   171		}
   172		return nil, ErrNotObj
   173	}

   177	// Set - Set the value of a field at a JSON path, any parts of the path that do not exist will be
   178	// constructed, and if a collision occurs with a non object type whilst iterating the path an error
   179	// is returned.
   180	func (g *Container) Set(value interface{}, path ...string) (*Container, error) {
   181		if len(path) == 0 {
   182			g.object = value
   183			return g, nil
   184		}
   185		var object interface{}
   186		if g.object == nil {
   187			g.object = map[string]interface{}{}
   188		}
   189		object = g.object
   190		for target := 0; target < len(path); target++ {
   191			if mmap, ok := object.(map[string]interface{}); ok {
   192				if target == len(path)-1 {
   193					mmap[path[target]] = value
   194				} else if mmap[path[target]] == nil {
   195					mmap[path[target]] = map[string]interface{}{}
   196				}
   197				object = mmap[path[target]]
   198			} else {
   199				return &Container{nil}, ErrPathCollision
   200			}
   201		}
   202		return &Container{object}, nil
   203	}

   210	// SetIndex - Set a value of an array element based on the index.
   211	func (g *Container) SetIndex(value interface{}, index int) (*Container, error) {
   212		if array, ok := g.Data().([]interface{}); ok {
   213			if index >= len(array) {
   214				return &Container{nil}, ErrOutOfBounds
   215			}
   216			array[index] = value
   217			return &Container{array[index]}, nil
   218		}
   219		return &Container{nil}, ErrNotArray
   220	}

   275	// Delete - Delete an element at a JSON path, an error is returned if the element does not exist.
   276	func (g *Container) Delete(path ...string) error {
   277		var object interface{}
   278	
   279		if g.object == nil {
   280			return ErrNotObj
   281		}
   282		object = g.object
   283		for target := 0; target < len(path); target++ {
   284			if mmap, ok := object.(map[string]interface{}); ok {
   285				if target == len(path)-1 {
   286					if _, ok := mmap[path[target]]; ok {
   287						delete(mmap, path[target])
   288					} else {
   289						return ErrNotObj
   290					}
   291				}
   292				object = mmap[path[target]]
   293			} else {
   294				return ErrNotObj
   295			}
   296		}
   297		return nil
   298	}

   312	// ArrayAppend - Append a value onto a JSON array.
   313	func (g *Container) ArrayAppend(value interface{}, path ...string) error {
   314		array, ok := g.Search(path...).Data().([]interface{})
   315		if !ok {
   316			return ErrNotArray
   317		}
   318		array = append(array, value)
   319		_, err := g.Set(array, path...)
   320		return err
   321	}

   328	// ArrayRemove - Remove an element from a JSON array.
   329	func (g *Container) ArrayRemove(index int, path ...string) error {
   330		if index < 0 {
   331			return ErrOutOfBounds
   332		}
   333		array, ok := g.Search(path...).Data().([]interface{})
   334		if !ok {
   335			return ErrNotArray
   336		}
   337		if index < len(array) {
   338			array = append(array[:index], array[index+1:]...)
   339		} else {
   340			return ErrOutOfBounds
   341		}
   342		_, err := g.Set(array, path...)
   343		return err
   344	}

   351	// ArrayElement - Access an element from a JSON array.
   352	func (g *Container) ArrayElement(index int, path ...string) (*Container, error) {
   353		if index < 0 {
   354			return &Container{nil}, ErrOutOfBounds
   355		}
   356		array, ok := g.Search(path...).Data().([]interface{})
   357		if !ok {
   358			return &Container{nil}, ErrNotArray
   359		}
   360		if index < len(array) {
   361			return &Container{array[index]}, nil
   362		}
   363		return &Container{nil}, ErrOutOfBounds
   364	}

   386	// Bytes - Converts the contained object back to a JSON []byte blob.
   387	func (g *Container) Bytes() []byte {
   388		if g.object != nil {
   389			if bytes, err := json.Marshal(g.object); err == nil {
   390				return bytes
   391			}
   392		}
   393		return []byte("{}")
   394	}
   395	
   396	// BytesIndent - Converts the contained object to a JSON []byte blob formatted with prefix, indent.
   397	func (g *Container) BytesIndent(prefix string, indent string) []byte {
   398		if g.object != nil {
   399			if bytes, err := json.MarshalIndent(g.object, prefix, indent); err == nil {
   400				return bytes
   401			}
   402		}
   403		return []byte("{}")
   404	}

   426	// ParseJSON - Convert a string into a representation of the parsed JSON.
   427	func ParseJSON(sample []byte) (*Container, error) {
   428		var gabs Container
   429	
   430		if err := json.Unmarshal(sample, &gabs.object); err != nil {
   431			return nil, err
   432		}
   433	
   434		return &gabs, nil
   435	}

   437	// ParseJSONDecoder - Convert a json.Decoder into a representation of the parsed JSON.
   438	func ParseJSONDecoder(decoder *json.Decoder) (*Container, error) {
   439		var gabs Container
   440	
   441		if err := decoder.Decode(&gabs.object); err != nil {
   442			return nil, err
   443		}

   445		return &gabs, nil
   446	}

   448	// ParseJSONFile - Read a file and convert into a representation of the parsed JSON.
   449	func ParseJSONFile(path string) (*Container, error) {
   450		if len(path) > 0 {
   451			cBytes, err := ioutil.ReadFile(path)
   452			if err != nil {
   453				return nil, err
   454			}
   455	
   456			container, err := ParseJSON(cBytes)
   457			if err != nil {
   458				return nil, err
   459			}
   460	
   461			return container, nil
   462		}
   463		return nil, ErrInvalidPath
   464	}
   465	
   466	// ParseJSONBuffer - Read the contents of a buffer into a representation of the parsed JSON.
   467	func ParseJSONBuffer(buffer io.Reader) (*Container, error) {
   468		var gabs Container
   469		jsonDecoder := json.NewDecoder(buffer)
   470		if err := jsonDecoder.Decode(&gabs.object); err != nil {
   471			return nil, err
   472		}
   473	
   474		return &gabs, nil
   475	}
