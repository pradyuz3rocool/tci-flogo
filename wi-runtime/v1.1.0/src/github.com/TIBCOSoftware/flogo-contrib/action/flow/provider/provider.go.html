    15	// Provider is the interface that describes an object
    16	// that can provide flow definitions from a URI
    17	type Provider interface {
    18	
    19		// GetFlow retrieves the flow definition for the specified id
    20		GetFlow(flowId string) (*definition.Definition, error)
    21		// AddCompressedFlow adds the flow for a specified id
    22		AddCompressedFlow(id string, flow string) error
    23		// AddUnCompressedFlow adds the flow for a specified id
    24		AddUncompressedFlow(id string, flow []byte) error
    25		// AddFlowURI adds the flow for a specified uri
    26		AddFlowURI(id string, uri string) error
    27	}

    29	// RemoteFlowProvider is an implementation of FlowProvider service
    30	// that can access flowes via URI
    31	type RemoteFlowProvider struct {
    32		//todo: switch to LRU cache

    63	// GetFlow implements flow.Provider.GetFlow
    64	func (pps *RemoteFlowProvider) GetFlow(id string) (*definition.Definition, error) {
    65	
    66		// todo turn pps.flowCache to real cache
    67		if flow, ok := pps.flowCache[id]; ok {
    68			logger.Debugf("Accessing cached Flow: %s\n")
    69			return flow, nil
    70		}
    71	
    72		logger.Debugf("Getting Flow: %s\n", id)
    73	
    74		flowRep, err := pps.flowMgr.GetFlow(id)
    75		if err != nil {
    76			return nil, err
    77		}
    78	
    79		def, err := definition.NewDefinition(flowRep)
    80		if err != nil {
    81			errorMsg := fmt.Sprintf("Error unmarshalling flow '%s': %s", id, err.Error())
    82			logger.Errorf(errorMsg)
    83			return nil, fmt.Errorf(errorMsg)
    84		}
    85	
    86		//todo hack until we fully move over to new action implementation
    87		factory := definition.GetLinkExprManagerFactory()
    88	
    89		if factory == nil {
    90			factory = &fggos.GosLinkExprManagerFactory{}
    91		}
    92	
    93		def.SetLinkExprManager(factory.NewLinkExprManager(def))
    94	
    95		//synchronize
    96		pps.mutex.Lock()
    97		pps.flowCache[id] = def
    98		pps.mutex.Unlock()
    99	
   100		return def, nil
   101	
   102	}
