   192	// DoStep performs a single execution 'step' of the Flow Instance
   193	func (pi *Instance) DoStep() bool {
   194	
   195		hasNext := false
   196	
   197		pi.ResetChanges()
   198	
   199		pi.stepID++
   200	
   201		if pi.status == StatusActive {
   202	
   203			item, ok := pi.WorkItemQueue.Pop()
   204	
   205			if ok {
   206				logger.Debug("popped item off queue")
   207	
   208				workItem := item.(*WorkItem)
   209	
   210				pi.ChangeTracker.trackWorkItem(&WorkItemQueueChange{ChgType: CtDel, ID: workItem.ID, WorkItem: workItem})
   211	
   212				pi.execTask(workItem)
   213				hasNext = true
   214			} else {
   215				logger.Debug("queue emtpy")
   216			}
   217		}
   218	
   219		return hasNext
   220	}

   239

   249	// execTask executes the specified Work Item of the Flow Instance
   250	func (pi *Instance) execTask(workItem *WorkItem) {
   251	
   252		defer func() {
   253			if r := recover(); r != nil {
   254	
   255				err := fmt.Errorf("Unhandled Error executing task '%s' : %v\n", workItem.TaskData.task.Name(), r)
   256				logger.Error(err)
   257	
   258				// todo: useful for debugging
   259				logger.Debugf("StackTrace: %s", debug.Stack())
   260	
   261				pi.appendActivityErrorData(workItem.TaskData, activity.NewError(err.Error(), "", nil))
   262				if workItem.TaskData.taskEnv.ID != idEhTasEnv {
   263					//not already in global handler, so handle it
   264					pi.HandleGlobalError()
   265				}
   266			}
   267		}()
   268	
   269		taskData := workItem.TaskData
   270		taskBehavior := pi.FlowModel.GetTaskBehavior(taskData.task.TypeID())
   271	
   272		var done bool
   273		var doneCode int
   274		var err error
   275	
   276		//todo: should validate process activities
   277	
   278		if workItem.ExecType == EtEval {
   279	
   280			eval := true
   281	
   282			if taskData.HasAttrs() {
   283	
   284				err := applyInputMapper(pi, taskData)
   285	
   286				if err != nil {
   287					pi.appendMapperErrorData(err)
   288					pi.HandleGlobalError()
   289					return
   290				}
   291	
   292				eval = applyInputInterceptor(pi, taskData)
   293			}
   294	
   295			if eval {
   296				done, doneCode, err = pi.evalTask(taskBehavior, taskData, workItem.EvalCode)
   297			} else {
   298				done = true
   299			}
   300		} else {
   301			done, doneCode, err = taskBehavior.PostEval(taskData, workItem.EvalCode, nil)
   302		}
   303	
   304		if err != nil {
   305			pi.handleTaskError(taskBehavior, taskData, err)
   306			return
   307		}
   308	
   309		if done {
   310	
   311			if taskData.HasAttrs() {
   312				applyOutputInterceptor(pi, taskData)
   313	
   314				appliedMapper, err := applyOutputMapper(pi, taskData)
   315	
   316				if err != nil {
   317					pi.appendMapperErrorData(err)
   318					pi.HandleGlobalError()
   319					return
   320				}
   321	
   322				if !appliedMapper && !taskData.task.IsScope() {
   323	
   324					logger.Debug("Applying Default Output Mapping")
   325					//applyDefaultActivityOutputMappings(pi, taskData)
   326				}
   327			}
   328	
   329			pi.handleTaskDone(taskBehavior, taskData, doneCode)
   330		}
   331	}

   333	func (pi *Instance) evalTask(taskBehavior model.TaskBehavior, taskData *TaskData, evalCode int) (done bool, doneCode int, err error) {
   334	
   335		defer func() {
   336			if r := recover(); r != nil {
   337	
   338				err = fmt.Errorf("Unhandled Error evaluating task '%s' : %v\n", taskData.task.Name(), r)
   339				logger.Error(err)
   340	
   341				// todo: useful for debugging
   342				logger.Debugf("StackTrace: %s", debug.Stack())
   343	
   344				done = false
   345				doneCode = 0
   346			}
   347		}()
   348	
   349		done, doneCode, err = taskBehavior.Eval(taskData, evalCode)
   350	
   351		return done, doneCode, err
   352	}

   354	// handleTaskDone handles the completion of a task in the Flow Instance
   355	func (pi *Instance) handleTaskDone(taskBehavior model.TaskBehavior, taskData *TaskData, doneCode int) {
   356	
   357		notifyParent, childDoneCode, taskEntries, err := taskBehavior.Done(taskData, doneCode)
   358	
   359		if err != nil {
   360			pi.appendErrorData(err)
   361			pi.HandleGlobalError()
   362			return
   363		}
   364	
   365		task := taskData.Task()
   366	
   367		if notifyParent {
   368	
   369			parentTask := task.Parent()
   370	
   371			if parentTask != nil {
   372				parentTaskData := taskData.taskEnv.TaskDatas[parentTask.ID()]
   373				parentBehavior := pi.FlowModel.GetTaskBehavior(parentTask.TypeID())
   374				parentDone, parentDoneCode := parentBehavior.ChildDone(parentTaskData, task, childDoneCode)
   375	
   376				if parentDone {
   377					pi.handleTaskDone(parentBehavior, parentTaskData, parentDoneCode)
   378				}
   379	
   380			} else {
   381	
   382				//todo distinguish between error handler env and rootTaskEnv
   383	
   384				//Root Task is Done, so notify Flow
   385				flowBehavior := pi.FlowModel.GetFlowBehavior()
   386				flowBehavior.TasksDone(pi, childDoneCode)
   387				flowBehavior.Done(pi)
   388	
   389				pi.setStatus(StatusCompleted)
   390			}
   391		}
   392	
   393		if len(taskEntries) > 0 {
   394	
   395			for _, taskEntry := range taskEntries {
   396	
   397				logger.Debugf("execTask - TaskEntry: %v\n", taskEntry)
   398				taskToEnterBehavior := pi.FlowModel.GetTaskBehavior(taskEntry.Task.TypeID())
   399	
   400				enterTaskData, _ := taskData.taskEnv.FindOrCreateTaskData(taskEntry.Task)
   401	
   402				eval, evalCode := taskToEnterBehavior.Enter(enterTaskData, taskEntry.EnterCode)
   403	
   404				if eval {
   405					pi.scheduleEval(enterTaskData, evalCode)
   406				}
   407			}
   408		}
   409	
   410		taskData.taskEnv.releaseTask(task)
   411	}

   443	// handleTaskError handles the completion of a task in the Flow Instance
   444	func (pi *Instance) handleTaskError(taskBehavior model.TaskBehavior, taskData *TaskData, err error) {
   445	
   446		handled, taskEntry := taskBehavior.Error(taskData)
   447	
   448		if !handled {
   449			pi.appendActivityErrorData(taskData, err)
   450			if taskData.taskEnv.ID != idEhTasEnv {
   451				//not already in global handler, so handle it
   452				pi.HandleGlobalError()
   453			}
   454			return
   455		}
   456	
   457		//todo add error data for task to flow
   458	
   459		if taskEntry != nil {
   460	
   461			logger.Debugf("execTask - TaskEntry: %v\n", taskEntry)
   462			taskToEnterBehavior := pi.FlowModel.GetTaskBehavior(taskEntry.Task.TypeID())
   463	
   464			enterTaskData, _ := taskData.taskEnv.FindOrCreateTaskData(taskEntry.Task)
   465	
   466			eval, evalCode := taskToEnterBehavior.Enter(enterTaskData, taskEntry.EnterCode)
   467	
   468			if eval {
   469				pi.scheduleEval(enterTaskData, evalCode)
   470			}
   471		}
   472	
   473		task := taskData.Task()
   474		taskData.taskEnv.releaseTask(task)
   475	}

   478	func (pi *Instance) HandleGlobalError() {
   479	
   480		if pi.Flow.ErrorHandlerTask() != nil {
   481	
   482			ehTask := pi.Flow.ErrorHandlerTask()
   483	
   484			if pi.EhTaskEnv == nil {
   485				var taskEnv TaskEnv
   486				taskEnv.ID = idEhTasEnv
   487				taskEnv.Task = ehTask
   488				taskEnv.taskID = ehTask.ID()
   489				taskEnv.Instance = pi
   490				taskEnv.TaskDatas = make(map[string]*TaskData)
   491				taskEnv.LinkDatas = make(map[int]*LinkData)
   492	
   493				pi.EhTaskEnv = &taskEnv
   494			}
   495	
   496			ehTaskData := pi.EhTaskEnv.TaskDatas[ehTask.ID()]
   497	
   498			if ehTaskData == nil {
   499				ehTaskData = pi.EhTaskEnv.NewTaskData(ehTask)
   500			}
   501	
   502			//todo: should we clear out the existing workitem queue?
   503	
   504			pi.scheduleEval(ehTaskData, 0)
   505		} else {
   506	
   507			//todo: log error information
   508			pi.setStatus(StatusFailed)
   509		}
   510	}

   589		if te.Instance == nil {

   608	func (te *TaskEnv) FindOrCreateTaskData(task *definition.Task) (taskData *TaskData, created bool) {
   609	
   610		taskData, ok := te.TaskDatas[task.ID()]
   611	
   612		created = false
   613

   617			te.Instance.ChangeTracker.trackTaskData(&TaskDataChange{ChgType: CtAdd, ID: task.ID(), TaskData: taskData})
   618	
   619			created = true
   620		}
   621	
   622		return taskData, created

   652	// releaseTask cleans up TaskData in the task environment any of its dependencies.
   653	// This is called when a task is completed and can be discarded
   654	func (te *TaskEnv) releaseTask(task *definition.Task) {
   655		delete(te.TaskDatas, task.ID())
   656		te.Instance.ChangeTracker.trackTaskData(&TaskDataChange{ChgType: CtDel, ID: task.ID()})
   657	
   658		childTasks := task.ChildTasks()
   659	
   660		if len(childTasks) > 0 {
   661	
   662			for _, childTask := range childTasks {
   663				delete(te.TaskDatas, childTask.ID())
   664				te.Instance.ChangeTracker.trackTaskData(&TaskDataChange{ChgType: CtDel, ID: childTask.ID()})
   665			}
   666		}
   667	
   668		links := task.FromLinks()
   669	
   670		for _, link := range links {
   671			delete(te.LinkDatas, link.ID())
   672			te.Instance.ChangeTracker.trackLinkData(&LinkDataChange{ChgType: CtDel, ID: link.ID()})
   673		}
   674	}

   856	// EvalLink implements activity.ActivityContext.EvalLink method
   857	func (td *TaskData) EvalLink(link *definition.Link) (result bool, err error) {
   858	
   859		logger.Debugf("TaskContext.EvalLink: %d\n", link.ID())
   860	
   861		defer func() {
   862			if r := recover(); r != nil {
   863				logger.Warnf("Unhandled Error evaluating link '%s' : %v\n", link.ID(), r)
   864	
   865				// todo: useful for debugging
   866				logger.Debugf("StackTrace: %s", debug.Stack())
   867	
   868				if err != nil {
   869					err = fmt.Errorf("%v", r)
   870				}
   871			}
   872		}()
   873	
   874		mgr := td.taskEnv.Instance.Flow.GetLinkExprManager()
   875	
   876		if mgr != nil {
   877			result, err = mgr.EvalLinkExpr(link, td.taskEnv.Instance)
   878			return result, err
   879		}
   880	
   881		return true, nil
   882	}

   889	// EvalActivity implements activity.ActivityContext.EvalActivity method
   890	func (td *TaskData) EvalActivity() (done bool, evalErr error) {
   891	
   892		act := activity.Get(td.task.ActivityRef())
   893	
   894		//todo: if act == nil, return TaskDoesntHaveActivity error or something like that
   895	
   896		defer func() {
   897			if r := recover(); r != nil {
   898				logger.Warnf("Unhandled Error executing activity '%s'[%s] : %v\n", td.task.Name(), td.task.ActivityRef(), r)
   899	
   900				// todo: useful for debugging
   901				logger.Debugf("StackTrace: %s", debug.Stack())
   902	
   903				if evalErr == nil {
   904					evalErr = activity.NewError(fmt.Sprintf("%v", r), "", nil)
   905					done = false
   906				}
   907			}
   908			if evalErr != nil {
   909				logger.Errorf("Execution failed for Activity[%s] in Flow[%s] - %s", td.task.Name(), td.taskEnv.Instance.Name(), evalErr.Error())
   910			}
   911		}()
   912	
   913		done, evalErr = act.Eval(td)
   914	
   915		return done, evalErr
   916	}
