    36

    38	
    39			// check if this task as an interceptor
    40			taskInterceptor := pi.Interceptor.GetTaskInterceptor(taskData.task.ID())
    41	
    42			if taskInterceptor != nil {
    43	
    44				logger.Debug("Applying Interceptor")
    45	
    46				if len(taskInterceptor.Inputs) > 0 {
    47					// override input attributes
    48					for _, attribute := range taskInterceptor.Inputs {
    49	
    50						logger.Debugf("Overriding Attr: %s = %s", attribute.Name, attribute.Value)
    51	
    52						//todo: validation
    53						taskData.InputScope().SetAttrValue(attribute.Name, attribute.Value)
    54					}
    55				}
    56	
    57				// check if we should not evaluate the task
    58				return !taskInterceptor.Skip
    59			}
    60		}
    61	
    62		return true
    63	}

   167	// GetAttr implements Scope.GetAttr
   168	func (s *FixedTaskScope) GetAttr(attrName string) (attr *data.Attribute, exists bool) {
   169	
   170		if len(s.attrs) > 0 {
   171	
   172			attr, found := s.attrs[attrName]
   173	
   174			if found {
   175				return attr, true
   176			}
   177		}
   178	
   179		if s.task != nil {
   180	
   181			var attr *data.Attribute
   182			var found bool
   183	
   184			if s.isInput {
   185				attr, found = s.task.GetInputAttr(attrName)
   186			} else {
   187				attr, found = s.task.GetOutputAttr(attrName)
   188			}
   189	
   190			if !found {
   191				attr, found = s.refAttrs[attrName]
   192			}
   193	
   194			return attr, found
   195		}
   196	
   197		return nil, false
   198	}

   200	// SetAttrValue implements Scope.SetAttrValue
   201	func (s *FixedTaskScope) SetAttrValue(attrName string, value interface{}) error {
   202	
   203		if len(s.attrs) == 0 {
   204			s.attrs = make(map[string]*data.Attribute)
   205		}
   206	
   207		logger.Debugf("SetAttr: %s = %v\n", attrName, value)
   208	
   209		attr, found := s.attrs[attrName]
   210	
   211		if found {
   212			//todo handle errors
   213			coercedVal, _ := data.CoerceToValue(value, attr.Type)
   214			attr.Value = coercedVal
   215		} else {
   216			// look up reference for type
   217			attr, found = s.refAttrs[attrName]
   218			if found {
   219				coercedVal, _ := data.CoerceToValue(value, attr.Type)
   220				s.attrs[attrName] = data.NewAttribute(attrName, attr.Type, coercedVal)
   221			} else {
   222				logger.Debugf("SetAttr: Attr %s ref not found\n", attrName)
   223				logger.Debugf("SetAttr: refs %v\n", s.refAttrs)
   224			}
   225			//todo: else error
   226		}
   227	
   228		return nil
   229	}
